Constante M = 50

Registro Clientes
{
    Entero DNI
    Cadena Nombre
}

Void Carga (Clientes A[M])
Comienzo
    Entero i

    Para i desde 0 hasta M - 1

        Escribir "Ingresar DNI: "
        Leer A[i].DNI

        Escribir "Ingresar Nombre: "
        Leer A[i].Nombre

    FinPara
Fin

Void MostrarOrdenado (Clientes A[M])
Comienzo
    Entero i, j, valor

    Para i desde 1 hasta M

        valor = A[i].DNI
        j = i-1

        Mientras ((j >= 0) Y (valor < A[j].DNI))

            A[j+1] = A[i]
            j = j-1
            
        FinMientras
    FinPara

    Para i desde M - 1 hasta 0

        Escribir "DNI: ", A[i].DNI
        Escribir "Nombre: ", A[i].Nombre

    FinPara     
Fin

Cadena Buscar (Clientes A[M], Entero elemento)
Comienzo
    Entero Inferior, Superior, Medio
    Cadena Esta

    Inferior = 0
    Superior = M - 1
    Medio = (Inferior + Superior) / 2
    Esta = "No se encontro en la estructura de datos"

    Mientras ((Inferior <= Superior) Y (A[Medio].DNI != elemento))

        Si (elemento < A[Medio])
            Entonces
                Superior = Medio - 1

            Sino (elemento > A[Medio])
                Inferior = Medio + 1    
        FinSi
    FinMientras

    Si (Inferior <= Superior)
        Entonces
            Retorna (A[Medio].Nombre)

        Sino
            Retorna (Esta)    
    FinSi
Fin

--/ Algoritmo Principal --/
Comienzo
    Clientes datos[M]
    Entero DniBuscar

    Carga (datos)
    MostrarOrdenado (datos)

    Escribir "Ingresar DNI a buscar: "
    Leer DniBuscar

    Escribir "Nombre ", Buscar (datos, DniBuscar)
Fin
