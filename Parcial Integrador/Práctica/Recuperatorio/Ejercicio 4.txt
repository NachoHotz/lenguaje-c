Algoritmo Perfumeria
Constante N = 40

Registro Articulos
{
    Entero Codigo
    Cadena Nombre
    Real Precio, total
}

Registro Ventas
{
    Cadena nombreArticulo
    Entero cantidadVentas
}

void Carga (Articulos A[N])
Comienzo
    Entero i

    Para i desde 0 hasta N-1

        A[i].total = 0

        Escribir "Ingresar codigo de artículo: "
        Leer A[i].Codigo

        Escribir "Ingresar nombre del artículo: "
        Leer A[i].Nombre

        Escribir "Ingresar precio del artículo: "   
        Leer A[i].Precio

    FinPara

    Retorna ()
Fin

void cargaVentas (Articulos A[N], Ventas B[N])
Comienzo
    Entero i, inferior, superior, medio
    Cadena Nom

    Escribir"Ingresar nombre"
    Leer Nom

    inferior = 0
    superior = N-1
    medio = (inferior + superior) / 2

    Para i desde 0 hasta N-1

        B[i].nombreArticulo = A[i].Nombre
        B[i].cantidadVentas = 0

    FinPara

    Mientras ((inferior <= superior) Y (Nom != B[medio].nombreArticulo))

        Si (Nom < B[medio].nombreArticulo)
        Entonces

            superior = medio - 1

        Sino (Nom > B[medio].nombreArticulo)

            inferior = medio + 1

        FinSi

        medio = (inferior + superior) / 2

    FinMientras

    Si (inferior <= superior)
    Entonces
    
        Escribir "Ingresar cantidad de ventas"
        Leer B[medio].cantidadVentas   

    FinSi

    Retorna ()

Fin

void totalVentas (Articulos A[N], Ventas B[N])
Comienzo
    Entero i

    Para i desde 0 hasta N-1

        A[i].total = A[i].Precio * B[i].cantidadVentas

    FinPara

    Para i desde 0 hasta N-1

        Escribir "Total de dinero recaudado en ventas del articulo", i+1, ": ", A[i].total

    FinPara

    Retorna ()
Fin

Real promedioPrecio (Articulos A[N])
Comienzo
    Entero i
    Real totalAcumulado, promedio

    totalAcumulado = 0

    Para i desde 0 hasta N-1

        totalAcumulado = totalAcumulado + A[i].Precio
        
    FinPara

    promedio = totalAcumulado / N

    Retorna (promedio)
Fin

void masVendido (Articulos A[N], Ventas B[N])
Comienzo
    Entero i, ventaMaxima, ventaMaximaCodigo, precioMinimoCodigo
    Real precioMinimo

    ventaMaxima = B[0].cantidadVentas

    Para i desde 0 hasta N-1

        Si (B[i].cantidadVentas > ventaMaxima)
        Entonces

            ventaMaxima = B[i].cantidadVentas
            ventaMaximaCodigo = A[i].Codigo

        FinSi
    FinPara

    precioMinimo = A[0].Precio

    Para i desde 0 hasta N-1

        Si (A[i].Precio < precioMinimo)
        Entonces

            precioMinimo = A[i].Precio
            precioMinimoCodigo = A[i].Codigo

        FinSi
    FinPara

    Si (ventaMaximaCodigo == precioMinimoCodigo)
    Entonces

        Escribir "El articulo", ventaMaximaCodigo, "es el mas economico y mas vendido"

    FinSi

    Retorna ()
Fin

Entero nuevaEstructura (Articulos A[N], Nuevo B[N])
Comienzo
    Entero i, cantidadComponentes
    Real precioNuevo

    Escribir "Ingresar precio"
    Leer precioNuevo

    Para i desde 0 hasta N-1

        Si (A[i].Precio < precioNuevo)
        Entonces

            B[i].nombre = A[i].Nombre
            cantidadComponentes = cantidadComponentes + 1

        FinSi
    FinPara

    Retorna (cantidadComponentes)
Fin

--/ Algoritmo Principal --/
Comienzo
    Articulos datos[N]
    Ventas datos2[N]

    Carga (datos)
    cargaVentas (datos, datos2)
    totalVentas (datos, datos2)

    Escribir "El precio promedio es", promedioPrecio (datos) 

    masVendido (datos, datos2)

    Escribir "La dimension de la nueva estructura es", nuevaEstructura (datos, datos2)

    Retorna ()
Fin
